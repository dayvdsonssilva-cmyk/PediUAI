// src/modules/financial.js import { supabase } from '../lib/supabase.js' export async function loadFinancial(store, period) { const from = getPeriodStart(period) const { data: orders } = await supabase .from('orders') .select('*') .eq('store_id', store.id) .eq('status', 'done') .gte('created_at', from.toISOString()) .order('created_at', { ascending: false }) const total = orders?.reduce((s, o) => s + (o.total || 0), 0) ?? 0 const count = orders?.length ?? 0 const avg = count ? total / count : 0 document.getElementById('fin-stats').innerHTML = `
Faturamento
R$ ${total.toFixed(2).replace('.', ',')}
Pedidos
${count}
Ticket médio
R$ ${avg.toFixed(2).replace('.', ',')}
` const tbody = document.getElementById('fin-body') if (tbody) { tbody.innerHTML = orders?.map(o => { const t = new Date(o.created_at).toLocaleTimeString('pt-BR', { hour:'2-digit', minute:'2-digit' }) return ` ${t} #${o.id_display || o.id.slice(-4).toUpperCase()} ${o.customer_name} ${o.payment} R$ ${o.total.toFixed(2).replace('.', ',')} ` }).join('') ?? '' } } export function setPeriod(period, btn) { document.querySelectorAll('.pb').forEach(b => b.classList.remove('active')) btn.classList.add('active') const labels = { hoje: 'Hoje', semana: 'Esta semana', mes: 'Este mês', '90': 'Últimos 90 dias' } document.getElementById('fin-period-lbl').textContent = labels[period] || period loadFinancial(window.__store, period) } function getPeriodStart(period) { const now = new Date() if (period === 'hoje') { now.setHours(0,0,0,0); return now } if (period === 'semana') { now.setDate(now.getDate() - now.getDay()); now.setHours(0,0,0,0); return now } if (period === 'mes') { now.setDate(1); now.setHours(0,0,0,0); return now } now.setDate(now.getDate() - 90); return now }
